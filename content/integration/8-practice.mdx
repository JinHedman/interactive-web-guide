  </body>

      name: "styles.css",

      code: `body {
  font-family: system-ui, sans-serif;
}`
    },
    {
      name: "script.js",
      lang: "js",
      code: `const display = document.getElementById("count");
const minusBtn = document.getElementById("minus");
const plusBtn = document.getElementById("plus");
const resetBtn = document.getElementById("reset");
let count = 0;

function render() {
  display.textContent = count;
}

plusBtn.addEventListener("click", function () {
  count = count + 1;
  render();
});

minusBtn.addEventListener("click", function () {
  if (count > 0) {
    count = count - 1;
  }
  render();
});

resetBtn.addEventListener("click", function () {
  count = 0;
  render();
});

render();`
    }
  ]}
/>

  title="Fetch and render with a loading state"
  goal="Fetch a small JSON array from a public API, show a 'Loading…' placeholder first, then replace it with a rendered list — running in your own browser, where fetch works."
  steps={[

    "In script.js, write an async function loadPosts that grabs the list element.",

    "In a try block, await fetch(\"https://jsonplaceholder.typicode.com/posts?_limit=5\"), then await response.json() into a posts variable.",

  solution={[
    {
      name: "index.html",
      lang: "html",
      code: `<!doctype html>
<html lang="en">
  <head>
    <meta charset="utf-8">
    <meta name="viewport" content="width=device-width, initial-scale=1">
    <title>Posts</title>
    <link rel="stylesheet" href="styles.css">
  </head>
  <body>
    <h1>Latest posts</h1>
    <ul id="list"></ul>

    <script src="script.js"></script>
  </body>
</html>`
    },
    {
      name: "styles.css",
      lang: "css",
      code: `body {
  font-family: system-ui, sans-serif;
  max-width: 36rem;
  margin: 2rem auto;
}
li {
  padding: 0.4rem 0;
  border-bottom: 1px solid #eee;
}`
    },
    {
      name: "script.js",
      lang: "js",
      code: `async function loadPosts() {
  const list = document.getElementById("list");
  list.textContent = "Loading…";
  try {
    const response = await fetch("https://jsonplaceholder.typicode.com/posts?_limit=5");
    const posts = await response.json();

    list.textContent = "";
    posts.forEach(function (post) {
      const li = document.createElement("li");
      li.textContent = post.title;
      list.append(li);
    });
  } catch (error) {
    list.textContent = "Could not load posts. Please try again.";
  }
}

loadPosts();`
    }
  ]}
/>
